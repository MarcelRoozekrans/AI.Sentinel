```csharp
// src/AI.Sentinel.Mcp/Logging/StderrLogger.cs
using System.Text;
using System.Text.Json;

namespace AI.Sentinel.Mcp.Logging;

/// <summary>Centralised stderr logger for the MCP proxy. Supports key=value (default) and NDJSON output.</summary>
internal static class StderrLogger
{
    private static bool UseJson => string.Equals(
        Environment.GetEnvironmentVariable("SENTINEL_MCP_LOG_JSON"),
        "1",
        StringComparison.Ordinal);

    /// <summary>Builds the stderr line for the supplied fields. Public for testing.</summary>
    public static string Format(IReadOnlyDictionary<string, string> fields)
    {
        if (UseJson)
            return JsonSerializer.Serialize(fields);

        var sb = new StringBuilder();
        var first = true;
        foreach (var kvp in fields)
        {
            if (!first) sb.Append(' ');
            sb.Append(kvp.Key).Append('=').Append(kvp.Value);
            first = false;
        }
        return sb.ToString();
    }

    /// <summary>Writes a stderr log line in the configured format.</summary>
    public static void Log(IReadOnlyDictionary<string, string> fields)
        => Console.Error.WriteLine(Format(fields));
}
```

### Step 4: Implement `ResourceReadInterceptor`

```csharp
// src/AI.Sentinel.Mcp/ResourceReadInterceptor.cs
using System.Text;
using AI.Sentinel.Audit;
using AI.Sentinel.Detection;
using AI.Sentinel.Mcp.Logging;
using Microsoft.Extensions.AI;
using ModelContextProtocol.Protocol;
using ModelContextProtocol.Server;

namespace AI.Sentinel.Mcp;

/// <summary>Intercepts <c>resources/read</c> responses, scans matching content items via <see cref="SentinelPipeline"/>.</summary>
internal static class ResourceReadInterceptor
{
    private static readonly string[] DefaultMimes =
    [
        "text/", "application/json", "application/xml", "application/yaml",
    ];

    public static Func<RequestContext<ReadResourceRequestParams>, CancellationToken, ValueTask<ReadResourceResult>> Create(
        SentinelPipeline pipeline,
        IAuditStore audit,
        Func<RequestContext<ReadResourceRequestParams>, CancellationToken, ValueTask<ReadResourceResult>> next)
    {
        var allowedMimes = ParseMimeAllowlist();
        var maxBytes = ParseMaxBytes();

        return async (ctx, ct) =>
        {
            var result = await next(ctx, ct).ConfigureAwait(false);
            if (result.Contents.Count == 0) return result;

            foreach (var content in result.Contents)
            {
                if (content.Text is not { Length: > 0 } text) continue; // blob — skip
                if (!IsAllowedMime(content.MimeType, allowedMimes))
                {
                    StderrLogger.Log(new Dictionary<string, string>(StringComparer.Ordinal)
                    {
                        ["event"]  = "resources_read",
                        ["action"] = "skipped",
                        ["reason"] = "mime",
                        ["mime"]   = content.MimeType ?? "(none)",
                    });
                    continue;
                }
                if (Encoding.UTF8.GetByteCount(text) > maxBytes)
                {
                    StderrLogger.Log(new Dictionary<string, string>(StringComparer.Ordinal)
                    {
                        ["event"]  = "resources_read",
                        ["action"] = "skipped",
                        ["reason"] = "oversize",
                        ["bytes"]  = Encoding.UTF8.GetByteCount(text).ToString(System.Globalization.CultureInfo.InvariantCulture),
                    });
                    continue;
                }

                try
                {
                    var pipelineResult = await pipeline.ScanMessagesAsync(
                        [MessageBuilder.BuildResourceRead(text)], ct).ConfigureAwait(false);
                    if (pipelineResult.MaxSeverity >= Severity.High)
                    {
                        // Mirror ToolCallInterceptor's block path
                        throw new ModelContextProtocol.McpProtocolException(
                            ModelContextProtocol.McpErrorCode.InternalError,
                            $"resources/read blocked: {pipelineResult.Detections[0].Summary}");
                    }
                }
                catch (ModelContextProtocol.McpProtocolException) { throw; }
                catch (Exception ex)
                {
                    StderrLogger.Log(new Dictionary<string, string>(StringComparer.Ordinal)
                    {
                        ["event"]  = "resources_read",
                        ["action"] = "fail_open",
                        ["error"]  = ex.GetType().Name,
                    });
                }
            }

            return result;
        };
    }

    private static HashSet<string> ParseMimeAllowlist()
    {
        var raw = Environment.GetEnvironmentVariable("SENTINEL_MCP_SCAN_MIMES");
        if (string.IsNullOrWhiteSpace(raw)) return new HashSet<string>(DefaultMimes, StringComparer.Ordinal);
        var set = new HashSet<string>(StringComparer.Ordinal);
        foreach (var m in raw.Split(',', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries))
            set.Add(m);
        return set;
    }

    private static int ParseMaxBytes()
    {
        var raw = Environment.GetEnvironmentVariable("SENTINEL_MCP_MAX_SCAN_BYTES");
        return int.TryParse(raw, System.Globalization.NumberStyles.Integer,
                            System.Globalization.CultureInfo.InvariantCulture, out var v) ? v : 65_536;
    }

    private static bool IsAllowedMime(string? mime, HashSet<string> allowed)
    {
        if (string.IsNullOrWhiteSpace(mime)) return false;
        foreach (var pattern in allowed)
        {
            // Wildcard suffix support: "text/" matches "text/plain", "text/html", etc.
            if (pattern.EndsWith('/'))
            {
                if (mime.StartsWith(pattern, StringComparison.OrdinalIgnoreCase)) return true;
            }
            else if (string.Equals(pattern, mime, StringComparison.OrdinalIgnoreCase))
            {
                return true;
            }
        }
        return false;
    }
}
```

> **Adapt the actual `ReadResourceResult` / `ResourceContents` types** to whatever the SDK exposes — names like `Contents`, `Text`, `Blob`, `MimeType` may vary slightly. Read the SDK types in your IDE if unsure.

```csharp
internal static IDictionary<string, string> ParseHttpHeaders(string? raw)

    var d = new Dictionary<string, string>(StringComparer.Ordinal);
```

```csharp
// In McpProxy.cs
internal static TimeSpan GetShutdownGrace()
{
    var raw = Environment.GetEnvironmentVariable("SENTINEL_MCP_TIMEOUT_SEC");
    return TimeSpan.FromSeconds(int.TryParse(raw, out var v) && v > 0 ? v : 5);
}

internal static async Task DisposeWithGraceAsync(IAsyncDisposable transport, TimeSpan grace)
{
    var disposeTask = transport.DisposeAsync().AsTask();
    if (await Task.WhenAny(disposeTask, Task.Delay(grace)).ConfigureAwait(false) != disposeTask)
    {
        StderrLogger.Log(new Dictionary<string, string>(StringComparer.Ordinal)
        {
            ["event"]    = "transport_dispose",
            ["action"]   = "grace_expired",
            ["grace_s"]  = grace.TotalSeconds.ToString("0", System.Globalization.CultureInfo.InvariantCulture),
        });

        // Simplest v1.1 path: log and proceed; the process will eventually be reaped by parent.
    }
}
```
